use zachapma

create TABLE TEAM(
TeamID int,
TeamName char(50) NOT NULL,
Nickname char(50) NOT NULL,
Rank int,
PRIMARY KEY(TeamID)
);

create TABLE GAME(
GameID int,
Rank1 int,
Rank2 int,
Location char(50),
Date date NOT NULL,
PRIMARY KEY(GameID)
);

create TABLE RESULT(
GameID int,
TeamOneID int,
TeamTwoID int,
ScoreOne int,
ScoreTwo int,
PRIMARY KEY(GameID),
FOREIGN KEY(GameID) REFERENCES GAME(GameID),
FOREIGN KEY(TeamOneID) REFERENCES TEAM(TeamID),
FOREIGN KEY(TeamTwoID) REFERENCES TEAM(TeamID)
);
